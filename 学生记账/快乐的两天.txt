$.ajax({
    url: "user/getSolrTest",//方法路径URL
    data: {
        name1:$('#solrtest').val()
    },//参数

    type: 'POST',
    async: false,//默认异步调用 false：同步

        //提交成功
        if (retInfo == true) {

        }
    },
    error: function () {
        alert('提交数据失败！');
    }
})
